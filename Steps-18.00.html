Create DAX Measures 18


1.      Go to Modeling tab on Ribbon 
2.      Click on New Measure 
3.      Fill in all the DAX formulas provided below, one at a time, and when you finish Press Enter or select Commit (checkmark icon) in the formula bar.  
4.      After creating each measure choose the necessary format and decimal for each value (measure)
5.      Save

DAX Formulas:

  • .ActiveEmployee = SUM('City-Parish_Employees'[Employee Cnt])-(SUM('City-Parish_Employees'[Attrition Count]))

  • .AttritionRate = DIVIDE(SUM('City-Parish_Employees'[Attrition Count]), (SUM('City-Parish_Employees'[Employee Cnt])))

  • .AttritionRateColor = IF([.AttritionRate]>0.3,"#D6272C","#1AAB40")

  • .AverageHourlyRate = AVERAGE('City-Parish_Employees'[HOURLY RATE])

  • .AverageYearOfService = AVERAGE('City-Parish_Employees'[YEARS OF SERVICE])

  • .AvgSalaryExpense = AVERAGE('City-Parish_Employees'[ANNUAL SALARY])

  • .InactiveEmployees = CALCULATE(COUNTROWS('City-Parish_Employees'), 'City-Parish_Employees'[EMPLOYMENT STATUS] = "I")
  • .TotalEmployees = SUM('City-Parish_Employees'[Employee Cnt])
  • .YOYActiveEmpArrow = SWITCH(TRUE(), [.YoYActiveEmployee] < 0, UNICHAR(9660),  // Down arrow[.YoYActiveEmployee] >= 0, UNICHAR(9650)  // Up arrow)

  • .YoYActiveEmployee = VAR CurrentPeriod = SUM('City Parish_Employees'[Employee Cnt]) - SUM('City-Parish_Employees'[Attrition Count]) VAR LastYearPeriod = CALCULATE(SUM('City-Parish_Employees'[Employee Cnt]) - SUM('City-Parish_Employees'[Attrition Count]), SAMEPERIODLASTYEAR('Date'[Date])  -- Assuming you have a Date table) RETURN IF(NOT ISBLANK(LastYearPeriod), (CurrentPeriod - LastYearPeriod) / LastYearPeriod,  -- YoY change as a percentage BLANK())
  • .YoYAttritionRate = VAR CurrentPeriodAttrition = DIVIDE(SUM('City-Parish_Employees'[Attrition Count]), SUM('City-Parish_Employees'[Employee Cnt])) VAR LastYearAttrition = CALCULATE(DIVIDE(SUM('City-Parish_Employees'[Attrition Count]), SUM('City-Parish_Employees'[Employee Cnt])),SAMEPERIODLASTYEAR('Date'[Date])) RETURN IF(NOT ISBLANK(LastYearAttrition), (CurrentPeriodAttrition - LastYearAttrition) / LastYearAttrition,BLANK())
  • .YOYAttritionRateArrow = SWITCH(TRUE(), [.YoYAttritionRate] < 0, UNICHAR(9660),  // Down arrow [.YoYAttritionRate] >= 0, UNICHAR(9650)  // Up arrow)
  • .YoYAvgHourlyRate = VAR CurrentYearAvg = AVERAGE('City-Parish_Employees'[HOURLY RATE]) VAR LastYearAvg = CALCULATE(AVERAGE('City-Parish_Employees'[HOURLY RATE]), SAMEPERIODLASTYEAR('Date'[Date])) RETURN IF( LastYearAvg <> 0,  (CurrentYearAvg - LastYearAvg) / LastYearAvg, BLANK())

  • .YOYAvgHourlyRateArrow = SWITCH(TRUE(),  [.YoYAvgHourlyRate] < 0, UNICHAR(9660),  // Down arrow   [.YoYAvgHourlyRate] >= 0, UNICHAR(9650)  // Up arrow)
  • .YoYAvgSalaryExpense = VAR CurrentYearAvg = AVERAGE('City-Parish_Employees'[ANNUAL SALARY])VAR LastYearAvg = CALCULATE(AVERAGE('City-Parish_Employees'[ANNUAL SALARY]), SAMEPERIODLASTYEAR('Date'[Date]))RETURN IF( LastYearAvg <> 0,  (CurrentYearAvg - LastYearAvg) / LastYearAvg, BLANK())
  • .YOYAvgSalaryExpenseArrow = SWITCH(TRUE(), [.YoYAvgSalaryExpense] < 0, UNICHAR(9660),  // Down arrow[.YoYAvgSalaryExpense] >= 0, UNICHAR(9650)  // Up arrow)
  • .YoYAvgYearsofService = VAR CurrentYearAvg = AVERAGE('City-Parish_Employees'[YEARS OF SERVICE])VAR LastYearAvg = CALCULATE(AVERAGE('City-Parish_Employees'[YEARS OF SERVICE]), SAMEPERIODLASTYEAR('Date'[Date]))RETURN IF(LastYearAvg <> 0, (CurrentYearAvg - LastYearAvg) / LastYearAvg,  BLANK())
  • .YOYAvgYearsofServiceArrow = SWITCH(TRUE(),   [.YoYAvgYearsofService] < 0, UNICHAR(9660),  // Down arrow [.YoYAvgYearsofService] >= 0, UNICHAR(9650)  // Up arrow)
  • .YoYSalaryExpense = VAR CurrentSalary = SUM('City-Parish_Employees'[ANNUAL SALARY])VAR LastYearSalary =  CALCULATE( SUM('City-Parish_Employees'[ANNUAL SALARY]),SAMEPERIODLASTYEAR('Date'[Date])  -- Assuming you have a Date table )RETURN IF(NOT ISBLANK(LastYearSalary),(CurrentSalary - LastYearSalary) / LastYearSalary,  -- YoY change as a percentageBLANK())
  • .YOYTotalEmpArrow = SWITCH(TRUE(),  [.YoYTotalEmployee] < 0, UNICHAR(9660),  // Down arrow  [.YoYTotalEmployee] >= 0, UNICHAR(9650)  // Up arrow)
  • .YoYTotalEmployee = VAR CurrentPeriod = SUM('City-Parish_Employees'[Employee Cnt])VAR LastYearPeriod =   CALCULATE(
     SUM('City-Parish_Employees'[Employee Cnt]), SAMEPERIODLASTYEAR('Date'[Date])  -- Assuming you have a Date table )RETURN IF(NOT ISBLANK(LastYearPeriod), (CurrentPeriod - LastYearPeriod) / LastYearPeriod,  -- YoY change as a percentage BLANK())
